program:
        TROLLEY CONST_ID CONTAINS declarations* CHECKOUT statement* SEMI DONE SEMI

term:

        procedureApplication

        INTEGER
        DOUBLE

unary:

mult:
        unary ((MULT | DIV | EXPON) unary)*

relational:

        relational (CONJ relational)*

expression:
        conjunction (DISJUNCT conjunction)*
