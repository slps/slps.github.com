asm_stmtblock:
        "asm" asm_stm* end_struct

asm_stm:
        asmlabel_colon? asm_unlabeledstm_semi?

asmlabel_colon:
        asmlabel colon

asm_unlabeledstm_semi:
        asm_unlabeledstm ";"? NL

asmid:

        "@+"* id
        "@+"* anynumber
        "@+"* key

asmlabel:

        label_id

asm_expr:

asm_infix_expr:

asm_term:
