emulate-negation

Source: SLPS/topics/grammars/dart/spec-0.01/emulate-negation.xbgf

  • replace ( ".", any in MULTI_LINE_STRING);
  • replace ( ".", any in MULTI_LINE_COMMENT);
  • replace ( ("'" | NEWLINE), any in SINGLE_LINE_STRING);
  • replace ( (""" | NEWLINE), any in SINGLE_LINE_STRING);
  • replace ( ("~" ("\" | """ | "$" | NEWLINE)), any in STRING_CONTENT_DQ);
  • replace ( ("~" ("\" | "'" | "$" | NEWLINE)), any in STRING_CONTENT_SQ);
  • replace ( ("~" NEWLINE), any);
  • replace ( ("~" NEWLINE*), any*);
